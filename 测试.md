let files = dv.pages('"000-箱的构造/箱中金句"');
if (files.length === 0) {
  dv.paragraph("没有找到任何文件");
} else {
  let randomFile = files[Math.floor(Math.random() * files.length)];
  let fileName = randomFile.name
  fileName = fileName.toString();
  let fileContent = await app.vault.readRaw(dv.page(fileName).file.path);
  if (!fileContent || !fileContent.lines || fileContent.lines.length < 1) {
    dv.paragraph("选中的文件没有足够的行");
  } else {
    let line1 = fileContent.lines[0];
    let sentence = line1.filter(x => x.startsWith("#金句"))
    dv.paragraph(sentence);
  }
}
