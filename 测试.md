let files = dv.pages('"000-箱的构造/金句"');
if (files.length === 0) {
  dv.paragraph("没有找到任何文件");
  return;
}
let randomFile = files[Math.floor(Math.random() * files.length)];
let fileContent = dv.current(randomFile).file;
if (!fileContent || !fileContent.lines || fileContent.lines.length < 10) {
  dv.paragraph("选中的文件没有足够的行");
  return;
}
let line10 = fileContent.lines[9];
let sentence = line10.replace('#金句', '').trim();
dv.paragraph(sentence);
