// 获取 "000-箱的构成/箱中金句" 文件夹中的所有文件

let randomFile = folderFiles[Math.floor(Math.random() * folderFiles.length)];

// 读取选中文件的内容并将其分割成行
let content = await app.vault.read(randomFile.path);
let lines = content.split("\n");

// 如果文件有至少10行，输出第10行，否则输出错误信息
if (lines.length >= 10) {
	dv.paragraph(lines[9]);
} else {
	dv.paragraph("Selected file does not have 10 lines.");
}
