let files = dv.pages('"100-UTTU人物合辑"');
if (files.length === 0) {
  dv.paragraph("没有找到任何文件");
} else {
  let randomFile = files[Math.floor(Math.random() * files.length)];
  let fileContent = dv.current(randomFile).file;
  if (!fileContent || !fileContent.lines || fileContent.lines.length < 1) {
    dv.paragraph("选中的文件没有足够的行");
  } else {
    let line1 = fileContent.lines[0];
    let sentence = line1.replace('#金句', '').trim();
    dv.paragraph(sentence);
  }
}
