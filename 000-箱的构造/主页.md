return function View() {
    const quotePages = dc.useQuery('@page and path("000-箱的构造/箱中memos") and #金句');
    const [sentence, setSentence] = dc.useState("正在读取金句...");
    const randomPage = dc.useMemo(() => {
        if (quotePages.length === 0) return null;
        return quotePages[Math.floor(Math.random() * quotePages.length)];
    }, [quotePages]);

    dc.useEffect(() => {
        let cancelled = false;

        async function loadQuote() {
            if (!randomPage) {
                setSentence("没有找到任何文件");
                return;
            }

            const file = dc.app.vault.getFileByPath(randomPage.$path);
            if (!file) {
                setSentence("没有找到任何文件");
                return;
            }

            const content = await dc.core.read(file);
            const line = content.split("\n").find(item => item.includes("#金句"));
            const cleanSentence = line ? line.replace(/#[^ ]+/g, "").trim() : "";

            if (!cancelled) {
                setSentence(cleanSentence || "没有找到金句内容");
            }
        }

        loadQuote();
        return () => {
            cancelled = true;
        };
    }, [randomPage ? randomPage.$path : ""]);

    return (
        <dc.Callout type="dictum" title="" collapsible={false}>
            <center>{sentence}</center>
        </dc.Callout>
    );
}

const EXCLUDED_PREFIXES = [
    "000-箱的构造/插件",
    "000-箱的构造/templates",
    "000-箱的构造/script",
    "assets"
];

function toMillis(value) {
    if (!value) return 0;
    if (typeof value.toMillis === "function") return value.toMillis();
    return new Date(value).getTime();
}

function formatDate(value) {
    const date = value && typeof value.toJSDate === "function" ? value.toJSDate() : new Date(value);
    return new Intl.DateTimeFormat("zh-CN", {
        year: "numeric",
        month: "2-digit",
        day: "2-digit"
    }).format(date);
}

function dayGreeting() {
    const hour = new Date().getHours();
    if (hour < 6) return "夜深了";
    if (hour < 9) return "早上好";
    if (hour < 12) return "上午好";
    if (hour < 14) return "中午好";
    if (hour < 18) return "下午好";
    return "晚上好";
}

return function View() {
    const pages = dc.useQuery("@page");
    const allPages = Array.from(pages);
    const firstPage = dc.useMemo(() => {
        return allPages.slice().sort((a, b) => toMillis(a.$ctime) - toMillis(b.$ctime))[0];
    }, [pages]);

    const visiblePages = allPages.filter(page => {
        return !EXCLUDED_PREFIXES.some(prefix => page.$path.startsWith(prefix));
    });
    const rolePages = allPages.filter(page => {
        return page.$path.startsWith("100-UTTU人物合辑/")
            && page.$path !== "100-UTTU人物合辑/100-UTTU人物合辑.md"
            && page.$path !== "100-UTTU人物合辑/维尔汀｜Vertin.md";
    });
    const psyPages = allPages.filter(page => {
        return page.$path.startsWith("200-相从心生/")
            && page.$path !== "200-相从心生/200-相从心生.md";
    });
    const tagCount = new Set(visiblePages.flatMap(page => page.$tags || [])).size;

    if (!firstPage) {
        return (
            <dc.Callout type="noborder|banner" title={<span>&nbsp;</span>} collapsible={false}>
                没有找到任何笔记。
            </dc.Callout>
        );
    }

    const totalDays = Math.floor((Date.now() - toMillis(firstPage.$ctime)) / (60 * 60 * 24 * 1000));
    const today = new Intl.DateTimeFormat("zh-CN", {
        year: "numeric",
        month: "long",
        day: "numeric",
        weekday: "long"
    }).format(new Date());

    return (
        <dc.Callout type="noborder|banner" title={<span>&nbsp;</span>} collapsible={false}>
            <p>
                {dayGreeting()}，今天是{today}。第一张档案记录于<em>{formatDate(firstPage.$ctime)}</em>，到今天已有<mark>{totalDays}</mark>天，积累档案及笔记<mark>{allPages.length}</mark>篇；共收录<mark>{rolePages.length}</mark>位箱中伙伴、共撰写<mark>{psyPages.length}</mark>篇心相档案、有<mark>{tagCount}</mark>个相关标签。
            </p>
        </dc.Callout>
    );
}
